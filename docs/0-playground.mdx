import {baseUrl} from './0-url.mdx'

export const MPFlutterPlayground = ({source, height}) => (
    <iframe style={{width: "100%", height: height || "300px", borderWidth: "1px", borderColor: "#f0f0f0"}} src={`https://pub.mpflutter.com/playground/index.html?source=${(() => {
        if (baseUrl === 'http://localhost:3001') {
            return baseUrl + source.replace('/zh/', '/');
        }
        return baseUrl + source;
    })()}`} />
)
